41  /*
42  * The bucket type holds SLOT_PER_BUCKET key-value pairs, along with their
43  * partial keys and occupancy info. It uses aligned_storage arrays to store
44  * the keys and values to allow constructing and destroying key-value pairs
45  * in place. The lifetime of bucket data should be managed by the container.
46  * It is the user's responsibility to confirm whether the data they are
47  * accessing is live or not.
48  */

191  // Constructs live data in a bucket
192  template <typename K, typename... Args>
193  void setKV(size_type ind, size_type slot, partial_t p, K &&k,
194  Args &&... args) {
195  bucket &b = buckets_[ind];
196  assert(!b.occupied(slot));
197  b.partial(slot) = p;
198  b.occupied(slot) = true;
199  key_traits_::construct(key_allocator_, std::addressof(b.key(slot)),
200  std::forward<K>(k));
201  mapped_traits_::construct(mapped_allocator_, std::addressof(b.mapped(slot)),
202  std::forward<Args>(args)...);
203  }
204 
205  // Destroys live data in a bucket
206  void eraseKV(size_type ind, size_type slot) {
207  bucket &b = buckets_[ind];
208  assert(b.occupied(slot));
209  b.occupied(slot) = false;
210  key_traits_::destroy(key_allocator_, std::addressof(b.key(slot)));
211  mapped_traits_::destroy(mapped_allocator_, std::addressof(b.mapped(slot)));
212  }
213 
214  // Moves data between two buckets in the container
215  void moveKV(size_type dst_ind, size_type dst_slot, size_type src_ind,
216  size_type src_slot) {
217  bucket &dst = buckets_[dst_ind];
218  bucket &src = buckets_[src_ind];
219  assert(src.occupied(src_slot));
220  assert(!dst.occupied(dst_slot));
221  setKV(dst_ind, dst_slot, src.partial(src_slot), src.movable_key(src_slot),
222  std::move(src.mapped(src_slot)));
223  eraseKV(src_ind, src_slot);
224  }

226  // Destroys all the live data in the buckets
227  void clear() noexcept {
228  static_assert(
229  std::is_nothrow_destructible<key_type>::value &&
230  std::is_nothrow_destructible<mapped_type>::value,
231  "libcuckoo_bucket_container requires key and value to be nothrow "
232  "destructible");
233  for (size_type i = 0; i < size(); ++i) {
234  bucket &b = buckets_[i];
235  for (size_type j = 0; j < SLOT_PER_BUCKET; ++j) {
236  if (b.occupied(j)) {
237  eraseKV(i, j);
238  }
239  }
240  }
241  }

243  // Creates a new container of hashpower `new_hp`, transfers the old data into
244  // it, and destroys the old container. Will not shrink the container.
245  void resize(size_type new_hp) {
246  assert(new_hp >= hashpower());
247  typename bucket_traits_::pointer new_buckets =
248  transfer(new_hp, *this, std::true_type());
249  destroy_buckets();

288  void destroy_buckets() noexcept {
289  if (buckets_ == nullptr) {
290  return;
291  }
292  // The bucket default constructor is nothrow, so we don't have to
293  // worry about dealing with exceptions when constructing all the
294  // elements.
295  static_assert(std::is_nothrow_destructible<bucket>::value,
296  "libcuckoo_bucket_container requires bucket to be nothrow "
297  "destructible");

299  for (size_type i = 0; i < size(); ++i) {

318  template <bool B>
319  typename bucket_traits_::pointer transfer(
320  size_type dst_hp,
321  typename std::conditional<B, libcuckoo_bucket_container &,
322  const libcuckoo_bucket_container &>::type src,
323  std::integral_constant<bool, B> move) {
324  assert(dst_hp >= src.hashpower());
325  libcuckoo_bucket_container dst(dst_hp, get_allocator());
326  // Move/copy all occupied slots of the source buckets
327  for (size_t i = 0; i < src.size(); ++i) {
328  for (size_t j = 0; j < SLOT_PER_BUCKET; ++j) {
329  if (src.buckets_[i].occupied(j)) {
330  dst.move_or_copy(i, j, src.buckets_[i], j, move);
331  }
332  }
333  }
334  // Take away the pointer from `dst` and return it
335  typename bucket_traits_::pointer dst_pointer = dst.buckets_;
336  dst.buckets_ = nullptr;
337  return dst_pointer;
338  }
