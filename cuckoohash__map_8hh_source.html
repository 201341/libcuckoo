552  typename std::enable_if<
553  std::is_convertible<Updater, updater_type>::value,
554  void>::type upsert(K&& key, Updater fn, Args&&... val) {
555  const hash_value hv = hashed_key(key);
556  cuckoo_status st;
557  do {
558  auto b = snapshot_and_lock_two(hv);
559  size_t hp = get_hashpower();
560  st = cuckoo_update_fn(key, fn, hv.partial, b.i[0], b.i[1]);
561  if (st == ok) {
562  break;
563  }
564 
565  // We run an insert, since the update failed. Since we already have
566  // the locks, we don't run cuckoo_insert_loop immediately, to avoid
567  // releasing and re-grabbing the locks. Recall that the locks will
568  // be released at the end of this call to cuckoo_insert.
569  st = cuckoo_insert(hv, std::move(b), std::forward<K>(key),
570  std::forward<Args>(val)...);
571  if (st == failure_table_full) {
572  cuckoo_fast_double(hp);
573  // Retry until the insert doesn't fail due to expansion.
574  if (cuckoo_insert_loop(hv, std::forward<K>(key),
575  std::forward<Args>(val)...)) {
576  break;
577  }
578  // The only valid reason for failure is a duplicate key. In this
579  // case, we retry the entire upsert operation.
580  }
581  } while (st != ok);
582  }

813  // the containers as a two (i1 and i2) and a one (i3). The one will not be
814  // active if i3 shares a lock index with i1 or i2.

816  // throws hashpower_changed if it changed after taking the lock.
817  std::pair<TwoBuckets, OneBucket>
818  lock_three(const size_t hp, const size_t i1,
819  const size_t i2, const size_t i3) const {

821  lock_ind(i1), lock_ind(i2), lock_ind(i3)}};
822  // Lock in order.
823  if (l[2] < l[1]) std::swap(l[2], l[1]);
824  if (l[2] < l[0]) std::swap(l[2], l[0]);
825  if (l[1] < l[0]) std::swap(l[1], l[0]);
826  locks_[l[0]].lock();
827  check_hashpower(hp, l[0]);
828  if (l[1] != l[0]) {
829  locks_[l[1]].lock();
830  }
831  if (l[2] != l[1]) {
832  locks_[l[2]].lock();
833  }
834  return std::make_pair(
835  TwoBuckets{this, i1, i2},
836  OneBucket{
837  (lock_ind(i3) == lock_ind(i1) ||
838  lock_ind(i3) == lock_ind(i2)) ?
839  nullptr : this, i3});
840  }

1047  // slot_search searches for a cuckoo path using breadth-first search. It
1048  // starts with the i1 and i2 buckets, and, until it finds a bucket with an
1049  // empty slot, adds each slot of the bucket in the b_slot. If the queue runs
1050  // out of space, it fails.
1051  //
1052  // throws hashpower_changed if it changed during the search
1053  b_slot slot_search(const size_t hp, const size_t i1,
1054  const size_t i2) {
1055  b_queue q;
1056  // The initial pathcode informs cuckoopath_search which bucket the path
1057  // starts on
1058  q.enqueue(b_slot(i1, 0, 0));
1059  q.enqueue(b_slot(i2, 1, 0));
1060  while (!q.full() && !q.empty()) {
1061  b_slot x = q.dequeue();
1062  // Picks a (sort-of) random slot to start from
1063  size_t starting_slot = x.pathcode % slot_per_bucket;
1064  for (size_t i = 0; i < slot_per_bucket && !q.full();
1065  ++i) {
1066  size_t slot = (starting_slot + i) % slot_per_bucket;
1067  OneBucket ob = lock_one(hp, x.bucket);
1068  Bucket& b = buckets_[x.bucket];
1069  if (!b.occupied(slot)) {
1070  // We can terminate the search here
1071  x.pathcode = x.pathcode * slot_per_bucket + slot;
1072  return x;
1073  }
1074 
1075  // If x has less than the maximum number of path components,
1076  // create a new b_slot item, that represents the bucket we would
1077  // have come from if we kicked out the item at this slot.
1078  const partial_t partial = b.partial(slot);
1079  if (x.depth < MAX_BFS_PATH_LEN - 1) {
1080  b_slot y(alt_index(hp, partial, x.bucket),
1081  x.pathcode * slot_per_bucket + slot, x.depth+1);
1082  q.enqueue(y);
1083  }
1084  }
1085  }
1086  // We didn't find a short-enough cuckoo path, so the queue ran out of
1087  // space. Return a failure value.
1088  return b_slot(0, 0, -1);
1089  }

1091  // cuckoopath_search finds a cuckoo path from one of the starting buckets to
1092  // an empty slot in another bucket. It returns the depth of the discovered
1093  // cuckoo path on success, and -1 on failure. Since it doesn't take locks on
1094  // the buckets it searches, the data can change between this function and
1095  // cuckoopath_move. Thus cuckoopath_move checks that the data matches the
1096  // cuckoo path before changing it.
1097  //
1098  // throws hashpower_changed if it changed during the search.
1099  int cuckoopath_search(const size_t hp,
1100  CuckooRecords& cuckoo_path,
1101  const size_t i1, const size_t i2) {
1102  b_slot x = slot_search(hp, i1, i2);
1103  if (x.depth == -1) {
1104  return -1;
1105  }
1106  // Fill in the cuckoo path slots from the end to the beginning.
1107  for (int i = x.depth; i >= 0; i--) {
1108  cuckoo_path[i].slot = x.pathcode % slot_per_bucket;
1109  x.pathcode /= slot_per_bucket;
1110  }
1111  // Fill in the cuckoo_path buckets and keys from the beginning to the
1112  // end, using the final pathcode to figure out which bucket the path
1113  // starts on. Since data could have been modified between slot_search
1114  // and the computation of the cuckoo path, this could be an invalid
1115  // cuckoo_path.
1116  CuckooRecord& first = cuckoo_path[0];
1117  if (x.pathcode == 0) {
1118  first.bucket = i1;
1119  } else {
1120  assert(x.pathcode == 1);
1121  first.bucket = i2;
1122  }
1123  {
1124  const OneBucket ob = lock_one(hp, first.bucket);
1125  const Bucket& b = buckets_[first.bucket];
1126  if (!b.occupied(first.slot)) {
1127  // We can terminate here
1128  return 0;
1129  }
1130  first.hv = hashed_key(b.key(first.slot));
1131  }
1132  for (int i = 1; i <= x.depth; ++i) {
1133  CuckooRecord& curr = cuckoo_path[i];
1134  const CuckooRecord& prev = cuckoo_path[i-1];
1135  assert(prev.bucket == index_hash(hp, prev.hv.hash) ||
1136  prev.bucket == alt_index(hp, prev.hv.partial,
1137  index_hash(hp, prev.hv.hash)));
1138  // We get the bucket that this slot is on by computing the alternate
1139  // index of the previous bucket
1140  curr.bucket = alt_index(hp, prev.hv.partial, prev.bucket);
1141  const OneBucket ob = lock_one(hp, curr.bucket);
1142  const Bucket& b = buckets_[curr.bucket];
1143  if (!b.occupied(curr.slot)) {
1144  // We can terminate here
1145  return i;
1146  }
1147  curr.hv = hashed_key(b.key(curr.slot));
1148  }
1149  return x.depth;

1152  // cuckoopath_move moves keys along the given cuckoo path in order to make
1153  // an empty slot in one of the buckets in cuckoo_insert. Before the start of
1154  // this function, the two insert-locked buckets were unlocked in run_cuckoo.
1155  // At the end of the function, if the function returns true (success), then
1156  // both insert-locked buckets remain locked. If the function is
1157  // unsuccessful, then both insert-locked buckets will be unlocked.
1158  //
1159  // throws hashpower_changed if it changed during the move.
1160  bool cuckoopath_move(const size_t hp, CuckooRecords& cuckoo_path,
1161  size_t depth, TwoBuckets& b) {
1162  assert(!b.is_active());
1163  if (depth == 0) {
1164  // There is a chance that depth == 0, when try_add_to_bucket sees
1165  // both buckets as full and cuckoopath_search finds one empty. In
1166  // this case, we lock both buckets. If the slot that
1167  // cuckoopath_search found empty isn't empty anymore, we unlock them
1168  // and return false. Otherwise, the bucket is empty and insertable,
1169  // so we hold the locks and return true.
1170  const size_t bucket = cuckoo_path[0].bucket;
1171  assert(bucket == b.i[0] || bucket == b.i[1]);
1172  b = lock_two(hp, b.i[0], b.i[1]);
1173  if (!buckets_[bucket].occupied(cuckoo_path[0].slot)) {
1174  return true;
1175  } else {
1176  b.release();
1177  return false;
1178  }
1179  }
1180 
1181  while (depth > 0) {
1182  CuckooRecord& from = cuckoo_path[depth-1];
1183  CuckooRecord& to = cuckoo_path[depth];
1184  Bucket& fb = buckets_[from.bucket];
1185  const size_t fs = from.slot;
1186  Bucket& tb = buckets_[to.bucket];
1187  const size_t ts = to.slot;
1188  TwoBuckets twob;
1189  OneBucket extrab;
1190  if (depth == 1) {
1191  // Even though we are only swapping out of one of the original
1192  // buckets, we have to lock both of them along with the slot we
1193  // are swapping to, since at the end of this function, they both
1194  // must be locked. We store tb inside the extrab container so it
1195  // is unlocked at the end of the loop.
1196  std::tie(twob, extrab) = lock_three(hp, b.i[0], b.i[1],
1197  to.bucket);
1198  } else {
1199  twob = lock_two(hp, from.bucket, to.bucket);
1200  }
1201 
1202  // We plan to kick out fs, but let's check if it is still there;
1203  // there's a small chance we've gotten scooped by a later cuckoo. If
1204  // that happened, just... try again. Also the slot we are filling in
1205  // may have already been filled in by another thread, or the slot we
1206  // are moving from may be empty, both of which invalidate the swap.
1207  // We only need to check that the hash value is the same, because,
1208  // even if the keys are different and have the same hash value, then
1209  // the cuckoopath is still valid.
1210  if (hashed_key_only_hash(fb.key(fs)) != from.hv.hash ||
1211  tb.occupied(ts) || !fb.occupied(fs)) {
1212  return false;
1213  }
1214 
1215  Bucket::move_to_bucket(fb, fs, tb, ts);
1216  if (depth == 1) {
1217  // Hold onto the locks contained in twob
1218  b = std::move(twob);
1219  }
1220  depth--;
1221  }
1222  return true;
1223  }

1225  // run_cuckoo performs cuckoo hashing on the table in an attempt to free up
1226  // a slot on either of the insert buckets, which are assumed to be locked
1227  // before the start. On success, the bucket and slot that was freed up is
1228  // stored in insert_bucket and insert_slot. In order to perform the search
1229  // and the swaps, it has to release the locks, which can lead to certain
1230  // concurrency issues, the details of which are explained in the function.
1231  // If run_cuckoo returns ok (success), then the bucket container will be
1232  // active, otherwise it will not.
1233  cuckoo_status run_cuckoo(TwoBuckets& b, size_t &insert_bucket,
1234  size_t &insert_slot) {
1235  // We must unlock the buckets here, so that cuckoopath_search and
1236  // cuckoopath_move can lock buckets as desired without deadlock.
1237  // cuckoopath_move has to move something out of one of the original
1238  // buckets as its last operation, and it will lock both buckets and
1239  // leave them locked after finishing. This way, we know that if
1240  // cuckoopath_move succeeds, then the buckets needed for insertion are
1241  // still locked. If cuckoopath_move fails, the buckets are unlocked and
1242  // we try again. This unlocking does present two problems. The first is
1243  // that another insert on the same key runs and, finding that the key
1244  // isn't in the table, inserts the key into the table. Then we insert
1245  // the key into the table, causing a duplication. To check for this, we
1246  // search the buckets for the key we are trying to insert before doing
1247  // so (this is done in cuckoo_insert, and requires that both buckets are
1248  // locked). Another problem is that an expansion runs and changes the
1249  // hashpower, meaning the buckets may not be valid anymore. In this
1250  // case, the cuckoopath functions will have thrown a hashpower_changed
1251  // exception, which we catch and handle here.
1252  size_t hp = get_hashpower();
1253  assert(b.is_active());
1254  b.release();
1255  CuckooRecords cuckoo_path;
1256  bool done = false;
1257  try {
1258  while (!done) {
1259  const int depth = cuckoopath_search(hp, cuckoo_path, b.i[0], b.i[1]);
1260  if (depth < 0) {
1261  break;
1262  }
1263 
1264  if (cuckoopath_move(hp, cuckoo_path, depth, b)) {
1265  insert_bucket = cuckoo_path[0].bucket;
1266  insert_slot = cuckoo_path[0].slot;
1267  assert(insert_bucket == b.i[0] || insert_bucket == b.i[1]);
1268  assert(!locks_[lock_ind(b.i[0])].try_lock());
1269  assert(!locks_[lock_ind(b.i[1])].try_lock());
1270  assert(!buckets_[insert_bucket].occupied(insert_slot));
1271  done = true;
1272  break;
1273  }
1274  }
1275  } catch (hashpower_changed&) {
1276  // The hashpower changed while we were trying to cuckoo, which means
1277  // we want to retry. b.i[0] and b.i[1] should not be locked in this
1278  // case.
1279  return failure_under_expansion;
1280  }
1281  return done ? ok : failure;
1282  }

1464  // cuckoo_insert tries to insert the given key-value pair into an empty slot
1465  // in either of the buckets, performing cuckoo hashing if necessary. It
1466  // expects the locks to be taken outside the function, but they are released
1467  // here, since different scenarios require different handling of the locks.
1468  // Before inserting, it checks that the key isn't already in the table.
1469  // cuckoo hashing presents multiple concurrency issues, which are explained
1470  // in the function. If the insert fails, the key and value won't be
1471  // move-constructed, so they can be retried.
1472  template <typename K, typename... Args>
1473  cuckoo_status cuckoo_insert(const hash_value hv, TwoBuckets b,
1474  K&& key, Args&&... val) {
1475  int res1, res2;
1476  Bucket& b0 = buckets_[b.i[0]];
1477  if (!try_find_insert_bucket(hv.partial, key, b0, res1)) {
1478  return failure_key_duplicated;
1479  }
1480  Bucket& b1 = buckets_[b.i[1]];
1481  if (!try_find_insert_bucket(hv.partial, key, b1, res2)) {
1482  return failure_key_duplicated;
1483  }
1484  if (res1 != -1) {
1485  add_to_bucket(hv.partial, b0, b.i[0], res1,
1486  std::forward<K>(key),
1487  std::forward<Args>(val)...);
1488  return ok;
1489  }
1490  if (res2 != -1) {
1491  add_to_bucket(hv.partial, b1, b.i[1], res2,
1492  std::forward<K>(key),
1493  std::forward<Args>(val)...);
1494  return ok;
1495  }
1496 
1497  // we are unlucky, so let's perform cuckoo hashing.
1498  size_t insert_bucket = 0;
1499  size_t insert_slot = 0;
1500  cuckoo_status st = run_cuckoo(b, insert_bucket, insert_slot);
1501  if (st == failure_under_expansion) {
1502  // The run_cuckoo operation operated on an old version of the table,
1503  // so we have to try again. We signal to the calling insert method
1504  // to try again by returning failure_under_expansion.
1505  return failure_under_expansion;
1506  } else if (st == ok) {
1507  assert(!locks_[lock_ind(b.i[0])].try_lock());
1508  assert(!locks_[lock_ind(b.i[1])].try_lock());
1509  assert(!buckets_[insert_bucket].occupied(insert_slot));
1510  assert(insert_bucket == index_hash(get_hashpower(), hv.hash) ||
1511  insert_bucket == alt_index(
1512  get_hashpower(), hv.partial,
1513  index_hash(get_hashpower(), hv.hash)));
1514  // Since we unlocked the buckets during run_cuckoo, another insert
1515  // could have inserted the same key into either b.i[0] or b.i[1], so
1516  // we check for that before doing the insert.
1517  if (cuckoo_contains(key, hv.partial, b.i[0], b.i[1])) {
1518  return failure_key_duplicated;
1519  }
1520  add_to_bucket(hv.partial, buckets_[insert_bucket], insert_bucket,
1521  insert_slot, std::forward<K>(key),
1522  std::forward<Args>(val)...);
1523  return ok;
1524  }
1525  assert(st == failure);
1526  LIBCUCKOO_DBG("hash table is full (hashpower = %zu, hash_items = %zu,"
1527  "load factor = %.2f), need to increase hashpower\n",
1528  get_hashpower(), cuckoo_size(),
1529  cuckoo_loadfactor(get_hashpower()));
1530  return failure_table_full;
1531  }

1642  void move_buckets(size_t current_hp, size_t new_hp,
1643  size_t start_lock_ind, size_t end_lock_ind) {
1644  for (; start_lock_ind < end_lock_ind; ++start_lock_ind) {
1645  for (size_t bucket_i = start_lock_ind;
1646  bucket_i < hashsize(current_hp);
1647  bucket_i += locks_t::size()) {
1648  // By doubling the table size, the index_hash and alt_index of
1649  // each key got one bit added to the top, at position
1650  // current_hp, which means anything we have to move will either
1651  // be at the same bucket position, or exactly
1652  // hashsize(current_hp) later than the current bucket
1653  Bucket& old_bucket = buckets_[bucket_i];
1654  const size_t new_bucket_i = bucket_i + hashsize(current_hp);
1655  Bucket& new_bucket = buckets_[new_bucket_i];
1656  size_t new_bucket_slot = 0;
1657 
1658  // Move each item from the old bucket that needs moving into the
1659  // new bucket
1660  for (size_t slot = 0; slot < slot_per_bucket; ++slot) {
1661  if (!old_bucket.occupied(slot)) {
1662  continue;
1663  }
1664  const hash_value hv = hashed_key(old_bucket.key(slot));
1665  const size_t old_ihash = index_hash(current_hp, hv.hash);
1666  const size_t old_ahash = alt_index(
1667  current_hp, hv.partial, old_ihash);
1668  const size_t new_ihash = index_hash(new_hp, hv.hash);
1669  const size_t new_ahash = alt_index(
1670  new_hp, hv.partial, new_ihash);
1671  if ((bucket_i == old_ihash && new_ihash == new_bucket_i) ||
1672  (bucket_i == old_ahash && new_ahash == new_bucket_i)) {
1673  // We're moving the key from the old bucket to the new
1674  // one
1675  Bucket::move_to_bucket(
1676  old_bucket, slot, new_bucket, new_bucket_slot++);
1677  // Also update the lock counts, in case we're moving to
1678  // a different lock.
1679  --locks_[lock_ind(bucket_i)].elems_in_buckets;
1680  ++locks_[lock_ind(new_bucket_i)].elems_in_buckets;
1681  } else {
1682  // Check that we don't want to move the new key
1683  assert(
1684  (bucket_i == old_ihash && new_ihash == old_ihash) ||
1685  (bucket_i == old_ahash && new_ahash == old_ahash));
1686  }
1687  }
1688  }
1689  // Now we can unlock the lock, because all the buckets corresponding
1690  // to it have been unlocked
1691  locks_[start_lock_ind].unlock();
1692  }
1693  }

1695  // Executes the function over the given range split over num_threads threads
1696  template <class F>
1697  static void parallel_exec(size_t start, size_t end,
1698  size_t num_threads, F func) {
1699  size_t work_per_thread = (end - start) / num_threads;
1700  std::vector<std::thread> threads(num_threads);
1701  std::vector<std::exception_ptr> eptrs(num_threads, nullptr);
1702  for (size_t i = 0; i < num_threads - 1; ++i) {
1703  threads[i] = std::thread(func, start, start + work_per_thread,
1704  std::ref(eptrs[i]));
1705  start += work_per_thread;
1706  }
1707  threads[num_threads - 1] = std::thread(
1708  func, start, end, std::ref(eptrs[num_threads - 1]));
1709  for (std::thread& t : threads) {
1710  t.join();
1711  }
1712  for (std::exception_ptr& eptr : eptrs) {
1713  if (eptr) {
1714  std::rethrow_exception(eptr);
1715  }
1716  }
1717  }

1719  // cuckoo_fast_double will double the size of the table by taking advantage
1720  // of the properties of index_hash and alt_index. If the key's move
1721  // constructor is not noexcept, we use cuckoo_expand_simple, since that
1722  // provides a strong exception guarantee.
1723  cuckoo_status cuckoo_fast_double(size_t current_hp) {
1724  if (!std::is_nothrow_move_constructible<storage_value_type>::value) {
1725  LIBCUCKOO_DBG("%s", "cannot run cuckoo_fast_double because kv-pair "
1726  "is not nothrow move constructible");
1727  return cuckoo_expand_simple(current_hp + 1, true);
1728  }
1729  const size_t new_hp = current_hp + 1;
1730  const size_t mhp = maximum_hashpower();
1731  if (mhp != NO_MAXIMUM_HASHPOWER && new_hp > mhp) {
1732  throw libcuckoo_maximum_hashpower_exceeded(new_hp);
1733  }
1734 
1735  std::lock_guard<expansion_lock_t> l(expansion_lock_);
1736  if (get_hashpower() != current_hp) {
1737  // Most likely another expansion ran before this one could grab the
1738  // locks
1739  LIBCUCKOO_DBG("%s", "another expansion is on-going\n");
1740  return failure_under_expansion;
1741  }
1742 
1743  locks_.allocate(std::min(locks_t::size(), hashsize(new_hp)));
1744  auto unlocker = snapshot_and_lock_all();
1745  buckets_.resize(buckets_.size() * 2);
1746  set_hashpower(new_hp);
1747 
1748  // We gradually unlock the new table, by processing each of the buckets
1749  // corresponding to each lock we took. For each slot in an old bucket,
1750  // we either leave it in the old bucket, or move it to the corresponding
1751  // new bucket. After we're done with the bucket, we release the lock on
1752  // it and the new bucket, letting other threads using the new map
1753  // gradually. We only unlock the locks being used by the old table,
1754  // because unlocking new locks would enable operations on the table
1755  // before we want them. We also re-evaluate the partial key stored at
1756  // each slot, since it depends on the hashpower.
1757  const size_t locks_to_move = std::min(locks_t::size(),
1758  hashsize(current_hp));
1759  parallel_exec(0, locks_to_move, kNumCores(),
1760  [this, current_hp, new_hp]
1761  (size_t start, size_t end, std::exception_ptr& eptr) {
1762  try {
1763  move_buckets(current_hp, new_hp, start, end);
1764  } catch (...) {
1765  eptr = std::current_exception();
1766  }
1767  });
1768  parallel_exec(locks_to_move, locks_.allocated_size(), kNumCores(),
1769  [this](size_t i, size_t end, std::exception_ptr&) {
1770  for (; i < end; ++i) {
1771  locks_[i].unlock();
1772  }
1773  });
1774  // Since we've unlocked the buckets ourselves, we don't need the
1775  // unlocker to do it for us.
1776  unlocker.deactivate();
1777  return ok;
1778  }

1780  // cuckoo_expand_simple will resize the table to at least the given
1781  // new_hashpower. If is_expansion is true, new_hashpower must be greater
1782  // than the current size of the table. If it's false, then new_hashpower
1783  // must be less. When we're shrinking the table, if the current table
1784  // contains more elements than can be held by new_hashpower, the resulting
1785  // hashpower will be greater than new_hashpower. It needs to take all the
1786  // bucket locks, since no other operations can change the table during
1787  // expansion. Throws libcuckoo_maximum_hashpower_exceeded if we're expanding
1788  // beyond the maximum hashpower, and we have an actual limit.
1789  cuckoo_status cuckoo_expand_simple(size_t new_hp,
1790  bool is_expansion) {
1791  const size_t mhp = maximum_hashpower();
1792  if (mhp != NO_MAXIMUM_HASHPOWER && new_hp > mhp) {
1793  throw libcuckoo_maximum_hashpower_exceeded(new_hp);
1794  }
1795  const auto unlocker = snapshot_and_lock_all();
1796  const size_t hp = get_hashpower();
1797  if ((is_expansion && new_hp <= hp) ||
1798  (!is_expansion && new_hp >= hp)) {
1799  // Most likely another expansion ran before this one could grab the
1800  // locks
1801  LIBCUCKOO_DBG("%s", "another expansion is on-going\n");
1802  return failure_under_expansion;
1803  }

1805  // Creates a new hash table with hashpower new_hp and adds all
1806  // the elements from the old buckets.
1807  cuckoohash_map<Key, T, Hash, Pred, Alloc, slot_per_bucket> new_map(
1808  hashsize(new_hp) * slot_per_bucket,
1809  0.0, /* minimum load factor */
1810  NO_MAXIMUM_HASHPOWER
1811  );
1812 
1813  parallel_exec(
1814  0, hashsize(hp), kNumCores(),
1815  [this, &new_map]
1816  (size_t i, size_t end, std::exception_ptr& eptr) {
1817  try {
1818  for (; i < end; ++i) {
1819  for (size_t j = 0; j < slot_per_bucket; ++j) {
1820  if (buckets_[i].occupied(j)) {
1821  storage_value_type& kvpair = (
1822  buckets_[i].storage_kvpair(j));
1823  new_map.insert(std::move(kvpair.first),
1824  std::move(kvpair.second));
1825  }
1826  }
1827  }
1828  } catch (...) {
1829  eptr = std::current_exception();
1830  }
1831  });
1832 
1833  // Swap the current buckets vector with new_map's and set the hashpower.
1834  // This is okay, because we have all the locks, so nobody else should be
1835  // reading from the buckets array. Then the old buckets array will be
1836  // deleted when new_map is deleted. All the locks should be released by
1837  // the unlocker as well.
1838  std::swap(buckets_, new_map.buckets_);
1839  set_hashpower(new_map.hashpower_);
1840  return ok;
1841  }
