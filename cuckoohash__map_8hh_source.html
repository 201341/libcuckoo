765  // hash value, making sure the hashpower doesn't change before the locks are
766  // taken. Thus it ensures that the buckets and locks corresponding to the
767  // hash value will stay correct as long as the locks are held. It returns

1035  template <typename K, typename LOCK_T>
1036  table_position cuckoo_insert_loop(hash_value hv, TwoBuckets<LOCK_T> &b,
1037  K &key) {
1038  table_position pos;
1039  while (true) {
1040  assert(b.is_active());
1041  const size_type hp = hashpower();
1042  pos = cuckoo_insert(hv, b, key);
1043  switch (pos.status) {
1044  case ok:
1045  case failure_key_duplicated:
1046  return pos;
1047  case failure_table_full:
1048  // Expand the table and try again, re-grabbing the locks
1049  cuckoo_fast_double<LOCK_T, automatic_resize>(hp);
1050  case failure_under_expansion:
1051  b = snapshot_and_lock_two<LOCK_T>(hv);
1052  break;
1053  default:
1054  assert(false);
1055  }
1056  }
1057  }

1059  // cuckoo_insert tries to find an empty slot in either of the buckets to
1060  // insert the given key into, performing cuckoo hashing if necessary. It
1061  // expects the locks to be taken outside the function. Before inserting, it
1062  // checks that the key isn't already in the table. cuckoo hashing presents
1063  // multiple concurrency issues, which are explained in the function. The
1064  // following return states are possible:
1065  //
1066  // ok -- Found an empty slot, locks will be held on both buckets after the
1067  // function ends, and the position of the empty slot is returned
1068  //
1069  // failure_key_duplicated -- Found a duplicate key, locks will be held, and
1070  // the position of the duplicate key will be returned
1071  //
1072  // failure_under_expansion -- Failed due to a concurrent expansion
1073  // operation. Locks are released. No meaningful position is returned.
1074  //
1075  // failure_table_full -- Failed to find an empty slot for the table. Locks
1076  // are released. No meaningful position is returned.
1077  template <typename K, typename LOCK_T>
1078  table_position cuckoo_insert(const hash_value hv, TwoBuckets<LOCK_T> &b,
1079  K &key) {
1080  int res1, res2;
1081  Bucket &b1 = buckets_[b.first()];
1082  if (!try_find_insert_bucket(b1, res1, hv.partial, key)) {
1083  return table_position{b.first(), static_cast<size_type>(res1),
1084  failure_key_duplicated};
1085  }
1086  Bucket &b2 = buckets_[b.second()];
1087  if (!try_find_insert_bucket(b2, res2, hv.partial, key)) {
1088  return table_position{b.second(), static_cast<size_type>(res2),
1089  failure_key_duplicated};
1090  }
1091  if (res1 != -1) {
1092  return table_position{b.first(), static_cast<size_type>(res1), ok};
1093  }
1094  if (res2 != -1) {
1095  return table_position{b.second(), static_cast<size_type>(res2), ok};
1096  }
1097 
1098  // We are unlucky, so let's perform cuckoo hashing.
1099  size_type insert_bucket = 0;
1100  size_type insert_slot = 0;
1101  cuckoo_status st = run_cuckoo<LOCK_T>(b, insert_bucket, insert_slot);
1102  if (st == failure_under_expansion) {
1103  // The run_cuckoo operation operated on an old version of the table,
1104  // so we have to try again. We signal to the calling insert method
1105  // to try again by returning failure_under_expansion.
1106  return table_position{0, 0, failure_under_expansion};
1107  } else if (st == ok) {
1108  assert(!locks_[lock_ind(b.first())].try_lock(LOCK_T()));
1109  assert(!locks_[lock_ind(b.second())].try_lock(LOCK_T()));
1110  assert(!buckets_[insert_bucket].occupied(insert_slot));
1111  assert(insert_bucket == index_hash(hashpower(), hv.hash) ||
1112  insert_bucket == alt_index(hashpower(), hv.partial,
1113  index_hash(hashpower(), hv.hash)));
1114  // Since we unlocked the buckets during run_cuckoo, another insert
1115  // could have inserted the same key into either b.first() or
1116  // b.second(), so we check for that before doing the insert.
1117  table_position pos = cuckoo_find(key, hv.partial, b.first(), b.second());
1118  if (pos.status == ok) {
1119  pos.status = failure_key_duplicated;
1120  return pos;
1121  }
1122  return table_position{insert_bucket, insert_slot, ok};
1123  }
1124  assert(st == failure);
1125  LIBCUCKOO_DBG("hash table is full (hashpower = %zu, hash_items = %zu,"
1126  "load factor = %.2f), need to increase hashpower\n",
1127  hashpower(), size(), load_factor());
1128  return table_position{0, 0, failure_table_full};
1129  }

1144  // try_find_insert_bucket will search the bucket for the given key, and for
1145  // an empty slot. If the key is found, we store the slot of the key in
1146  // `slot` and return false. If we find an empty slot, we store its position
1147  // in `slot` and return true. If no duplicate key is found and no empty slot
1148  // is found, we store -1 in `slot` and return true.
1149  template <typename K>
1150  bool try_find_insert_bucket(const Bucket &b, int &slot,
1151  const partial_t partial, const K &key) const {
1152  // Silence a warning from MSVC about partial being unused if is_simple.
1153  (void)partial;
1154  slot = -1;
1155  for (size_type i = 0; i < slot_per_bucket(); ++i) {
1156  if (b.occupied(i)) {
1157  if (!is_simple && partial != b.partial(i)) {
1158  continue;
1159  }
1160  if (key_eq()(b.key(i), key)) {
1161  slot = i;
1162  return false;
1163  }
1164  } else {
1165  slot = i;
1166  }
1167  }
1168  return true;

1172  // elements only define a sequence of alternate hashings for different hash
1173  // values, we only need to keep track of the hash values being moved, rather
1174  // than the keys themselves.
1175  typedef struct {

1177  size_type slot;

1181  // The maximum number of items in a cuckoo BFS path.
1182  static constexpr uint8_t MAX_BFS_PATH_LEN = 5;

1184  // An array of CuckooRecords
1185  using CuckooRecords = std::array<CuckooRecord, MAX_BFS_PATH_LEN>;
1186 
1187  // run_cuckoo performs cuckoo hashing on the table in an attempt to free up
1188  // a slot on either of the insert buckets, which are assumed to be locked
1189  // before the start. On success, the bucket and slot that was freed up is
1190  // stored in insert_bucket and insert_slot. In order to perform the search
1191  // and the swaps, it has to release the locks, which can lead to certain
1192  // concurrency issues, the details of which are explained in the function.
1193  // If run_cuckoo returns ok (success), then `b` will be active, otherwise it
1194  // will not.
1195  template <typename LOCK_T>
1196  cuckoo_status run_cuckoo(TwoBuckets<LOCK_T> &b, size_type &insert_bucket,
1197  size_type &insert_slot) {
1198  // We must unlock the buckets here, so that cuckoopath_search and
1199  // cuckoopath_move can lock buckets as desired without deadlock.
1200  // cuckoopath_move has to move something out of one of the original
1201  // buckets as its last operation, and it will lock both buckets and
1202  // leave them locked after finishing. This way, we know that if
1203  // cuckoopath_move succeeds, then the buckets needed for insertion are
1204  // still locked. If cuckoopath_move fails, the buckets are unlocked and
1205  // we try again. This unlocking does present two problems. The first is
1206  // that another insert on the same key runs and, finding that the key
1207  // isn't in the table, inserts the key into the table. Then we insert
1208  // the key into the table, causing a duplication. To check for this, we
1209  // search the buckets for the key we are trying to insert before doing
1210  // so (this is done in cuckoo_insert, and requires that both buckets are
1211  // locked). Another problem is that an expansion runs and changes the
1212  // hashpower, meaning the buckets may not be valid anymore. In this
1213  // case, the cuckoopath functions will have thrown a hashpower_changed
1214  // exception, which we catch and handle here.
1215  size_type hp = hashpower();
1216  b.unlock();
1217  CuckooRecords cuckoo_path;
1218  bool done = false;
1219  try {
1220  while (!done) {
1221  const int depth =
1222  cuckoopath_search<LOCK_T>(hp, cuckoo_path, b.first(), b.second());
1223  if (depth < 0) {
1224  break;
1225  }
1226 
1227  if (cuckoopath_move(hp, cuckoo_path, depth, b)) {
1228  insert_bucket = cuckoo_path[0].bucket;
1229  insert_slot = cuckoo_path[0].slot;
1230  assert(insert_bucket == b.first() || insert_bucket == b.second());
1231  assert(!locks_[lock_ind(b.first())].try_lock(LOCK_T()));
1232  assert(!locks_[lock_ind(b.second())].try_lock(LOCK_T()));
1233  assert(!buckets_[insert_bucket].occupied(insert_slot));
1234  done = true;
1235  break;
1236  }
1237  }
1238  } catch (hashpower_changed &) {
1239  // The hashpower changed while we were trying to cuckoo, which means
1240  // we want to retry. b.first() and b.second() should not be locked
1241  // in this case.
1242  return failure_under_expansion;
1243  }
1244  return done ? ok : failure;
1245  }

1247  // cuckoopath_search finds a cuckoo path from one of the starting buckets to
1248  // an empty slot in another bucket. It returns the depth of the discovered
1249  // cuckoo path on success, and -1 on failure. Since it doesn't take locks on
1250  // the buckets it searches, the data can change between this function and
1251  // cuckoopath_move. Thus cuckoopath_move checks that the data matches the
1252  // cuckoo path before changing it.
1253  //
1254  // throws hashpower_changed if it changed during the search.
1255  template <typename LOCK_T>
1256  int cuckoopath_search(const size_type hp, CuckooRecords &cuckoo_path,
1257  const size_type i1, const size_type i2) {
1258  b_slot x = slot_search<LOCK_T>(hp, i1, i2);
1259  if (x.depth == -1) {
1260  return -1;
1261  }
1262  // Fill in the cuckoo path slots from the end to the beginning.
1263  for (int i = x.depth; i >= 0; i--) {
1264  cuckoo_path[i].slot = x.pathcode % slot_per_bucket();
1265  x.pathcode /= slot_per_bucket();
1266  }
1267  // Fill in the cuckoo_path buckets and keys from the beginning to the
1268  // end, using the final pathcode to figure out which bucket the path
1269  // starts on. Since data could have been modified between slot_search
1270  // and the computation of the cuckoo path, this could be an invalid
1271  // cuckoo_path.
1272  CuckooRecord &first = cuckoo_path[0];
1273  if (x.pathcode == 0) {
1274  first.bucket = i1;
1275  } else {
1276  assert(x.pathcode == 1);
1277  first.bucket = i2;
1278  }
1279  {
1280  const auto ob = lock_one<LOCK_T>(hp, first.bucket);
1281  const Bucket &b = buckets_[first.bucket];
1282  if (!b.occupied(first.slot)) {
1283  // We can terminate here
1284  return 0;
1285  }
1286  first.hv = hashed_key(b.key(first.slot));
1287  }
1288  for (int i = 1; i <= x.depth; ++i) {
1289  CuckooRecord &curr = cuckoo_path[i];
1290  const CuckooRecord &prev = cuckoo_path[i - 1];
1291  assert(prev.bucket == index_hash(hp, prev.hv.hash) ||
1292  prev.bucket ==
1293  alt_index(hp, prev.hv.partial, index_hash(hp, prev.hv.hash)));
1294  // We get the bucket that this slot is on by computing the alternate
1295  // index of the previous bucket
1296  curr.bucket = alt_index(hp, prev.hv.partial, prev.bucket);
1297  const auto ob = lock_one<LOCK_T>(hp, curr.bucket);
1298  const Bucket &b = buckets_[curr.bucket];
1299  if (!b.occupied(curr.slot)) {
1300  // We can terminate here
1301  return i;
1302  }
1303  curr.hv = hashed_key(b.key(curr.slot));
1304  }
1305  return x.depth;
1306  }

1308  // cuckoopath_move moves keys along the given cuckoo path in order to make
1309  // an empty slot in one of the buckets in cuckoo_insert. Before the start of
1310  // this function, the two insert-locked buckets were unlocked in run_cuckoo.
1311  // At the end of the function, if the function returns true (success), then
1312  // both insert-locked buckets remain locked. If the function is
1313  // unsuccessful, then both insert-locked buckets will be unlocked.
1314  //
1315  // throws hashpower_changed if it changed during the move.
1316  template <typename LOCK_T>
1317  bool cuckoopath_move(const size_type hp, CuckooRecords &cuckoo_path,
1318  size_type depth, TwoBuckets<LOCK_T> &b) {
1319  assert(!b.is_active());
1320  if (depth == 0) {
1321  // There is a chance that depth == 0, when try_add_to_bucket sees
1322  // both buckets as full and cuckoopath_search finds one empty. In
1323  // this case, we lock both buckets. If the slot that
1324  // cuckoopath_search found empty isn't empty anymore, we unlock them
1325  // and return false. Otherwise, the bucket is empty and insertable,
1326  // so we hold the locks and return true.
1327  const size_type bucket = cuckoo_path[0].bucket;
1328  assert(bucket == b.first() || bucket == b.second());
1329  b = lock_two<LOCK_T>(hp, b.first(), b.second());
1330  if (!buckets_[bucket].occupied(cuckoo_path[0].slot)) {
1331  return true;
1332  } else {
1333  b.unlock();
1334  return false;
1335  }
1336  }
1337 
1338  while (depth > 0) {
1339  CuckooRecord &from = cuckoo_path[depth - 1];
1340  CuckooRecord &to = cuckoo_path[depth];
1341  const size_type fs = from.slot;
1342  const size_type ts = to.slot;
1343  TwoBuckets<LOCK_T> twob;
1344  OneBucket<LOCK_T> extrab;
1345  if (depth == 1) {
1346  // Even though we are only swapping out of one of the original
1347  // buckets, we have to lock both of them along with the slot we
1348  // are swapping to, since at the end of this function, they both
1349  // must be locked. We store tb inside the extrab container so it
1350  // is unlocked at the end of the loop.
1351  std::tie(twob, extrab) =
1352  lock_three<LOCK_T>(hp, b.first(), b.second(), to.bucket);
1353  } else {
1354  twob = lock_two<LOCK_T>(hp, from.bucket, to.bucket);
1355  }
1356 
1357  Bucket &fb = buckets_[from.bucket];
1358  Bucket &tb = buckets_[to.bucket];
1359 
1360  // We plan to kick out fs, but let's check if it is still there;
1361  // there's a small chance we've gotten scooped by a later cuckoo. If
1362  // that happened, just... try again. Also the slot we are filling in
1363  // may have already been filled in by another thread, or the slot we
1364  // are moving from may be empty, both of which invalidate the swap.
1365  // We only need to check that the hash value is the same, because,
1366  // even if the keys are different and have the same hash value, then
1367  // the cuckoopath is still valid.
1368  if (hashed_key_only_hash(fb.key(fs)) != from.hv.hash || tb.occupied(ts) ||
1369  !fb.occupied(fs)) {
1370  return false;
1371  }
1372 
1373  Bucket::move_to_bucket(allocator_, fb, fs, tb, ts);
1374  if (depth == 1) {
1375  // Hold onto the locks contained in twob
1376  b = std::move(twob);
1377  }
1378  depth--;
1379  }
1380  return true;
1381  }

1463  // slot_search searches for a cuckoo path using breadth-first search. It
1464  // starts with the i1 and i2 buckets, and, until it finds a bucket with an
1465  // empty slot, adds each slot of the bucket in the b_slot. If the queue runs
1466  // out of space, it fails.
1467  //
1468  // throws hashpower_changed if it changed during the search
1469  template <typename LOCK_T>
1470  b_slot slot_search(const size_type hp, const size_type i1,
1471  const size_type i2) {
1472  b_queue q;
1473  // The initial pathcode informs cuckoopath_search which bucket the path
1474  // starts on
1475  q.enqueue(b_slot(i1, 0, 0));
1476  q.enqueue(b_slot(i2, 1, 0));
1477  while (!q.full() && !q.empty()) {
1478  b_slot x = q.dequeue();
1479  // Picks a (sort-of) random slot to start from
1480  size_type starting_slot = x.pathcode % slot_per_bucket();
1481  for (size_type i = 0; i < slot_per_bucket() && !q.full(); ++i) {
1482  size_type slot = (starting_slot + i) % slot_per_bucket();
1483  auto ob = lock_one<LOCK_T>(hp, x.bucket);
1484  Bucket &b = buckets_[x.bucket];
1485  if (!b.occupied(slot)) {
1486  // We can terminate the search here
1487  x.pathcode = x.pathcode * slot_per_bucket() + slot;
1488  return x;
1489  }
1490 
1491  // If x has less than the maximum number of path components,
1492  // create a new b_slot item, that represents the bucket we would
1493  // have come from if we kicked out the item at this slot.
1494  const partial_t partial = b.partial(slot);
1495  if (x.depth < MAX_BFS_PATH_LEN - 1) {
1496  b_slot y(alt_index(hp, partial, x.bucket),
1497  x.pathcode * slot_per_bucket() + slot, x.depth + 1);
1498  q.enqueue(y);
1499  }
1500  }
1501  }
1502  // We didn't find a short-enough cuckoo path, so the queue ran out of
1503  // space. Return a failure value.
1504  return b_slot(0, 0, -1);
1505  }

1507  // cuckoo_fast_double will double the size of the table by taking advantage
1508  // of the properties of index_hash and alt_index. If the key's move
1509  // constructor is not noexcept, we use cuckoo_expand_simple, since that
1510  // provides a strong exception guarantee.
1511  template <typename LOCK_T, typename AUTO_RESIZE>
1512  cuckoo_status cuckoo_fast_double(size_type current_hp) {
1513  if (!std::is_nothrow_move_constructible<storage_value_type>::value) {
1514  LIBCUCKOO_DBG("%s", "cannot run cuckoo_fast_double because kv-pair "
1515  "is not nothrow move constructible");
1516  return cuckoo_expand_simple<LOCK_T, AUTO_RESIZE>(current_hp + 1);
1517  }
1518  const size_type new_hp = current_hp + 1;
1519  std::lock_guard<expansion_lock_t> l(expansion_lock_);
1520  cuckoo_status st = check_resize_validity<AUTO_RESIZE>(current_hp, new_hp);
1521  if (st != ok) {
1522  return st;
1523  }
1524 
1525  locks_.resize(hashsize(new_hp));
1526  auto unlocker = snapshot_and_lock_all<LOCK_T>();
1527  // We can't just resize, since the Bucket is non-copyable and
1528  // non-movable. Instead, we allocate a new array of buckets, and move
1529  // the contents of each bucket manually.
1530  {
1531  buckets_t new_buckets(buckets_.size() * 2, get_allocator());
1532  for (size_type i = 0; i < buckets_.size(); ++i) {
1533  Bucket::move_bucket(allocator_, buckets_[i], new_buckets[i]);
1534  }
1535  buckets_.swap(new_buckets);
1536  }
1537  set_hashpower(new_hp);
1538 
1539  // We gradually unlock the new table, by processing each of the buckets
1540  // corresponding to each lock we took. For each slot in an old bucket,
1541  // we either leave it in the old bucket, or move it to the corresponding
1542  // new bucket. After we're done with the bucket, we release the lock on
1543  // it and the new bucket, letting other threads using the new map
1544  // gradually. We only unlock the locks being used by the old table,
1545  // because unlocking new locks would enable operations on the table
1546  // before we want them. We also re-evaluate the partial key stored at
1547  // each slot, since it depends on the hashpower.
1548  const size_type locks_to_move =
1549  std::min(locks_.size(), hashsize(current_hp));
1550  parallel_exec(0, locks_to_move,
1551  [this, current_hp, new_hp](size_type start, size_type end,
1552  std::exception_ptr &eptr) {
1553  try {
1554  move_buckets<LOCK_T>(current_hp, new_hp, start, end);
1555  } catch (...) {
1556  eptr = std::current_exception();
1557  }
1558  });
1559  parallel_exec(locks_to_move, locks_.size(),
1560  [this](size_type i, size_type end, std::exception_ptr &) {
1561  for (; i < end; ++i) {
1562  locks_[i].unlock(LOCK_T());
1563  }
1564  });
1565  // Since we've unlocked the buckets ourselves, we don't need the
1566  // unlocker to do it for us.
1567  unlocker.release();
1568  return ok;
1569  }

1571  template <typename LOCK_T>
1572  void move_buckets(size_type current_hp, size_type new_hp,
1573  size_type start_lock_ind, size_type end_lock_ind) {
1574  for (; start_lock_ind < end_lock_ind; ++start_lock_ind) {
1575  for (size_type bucket_i = start_lock_ind; bucket_i < hashsize(current_hp);
1576  bucket_i += locks_t::max_size()) {
1577  // By doubling the table size, the index_hash and alt_index of
1578  // each key got one bit added to the top, at position
1579  // current_hp, which means anything we have to move will either
1580  // be at the same bucket position, or exactly
1581  // hashsize(current_hp) later than the current bucket
1582  Bucket &old_bucket = buckets_[bucket_i];
1583  const size_type new_bucket_i = bucket_i + hashsize(current_hp);
1584  Bucket &new_bucket = buckets_[new_bucket_i];
1585  size_type new_bucket_slot = 0;
1586 
1587  // Move each item from the old bucket that needs moving into the
1588  // new bucket
1589  for (size_type slot = 0; slot < slot_per_bucket(); ++slot) {
1590  if (!old_bucket.occupied(slot)) {
1591  continue;
1592  }
1593  const hash_value hv = hashed_key(old_bucket.key(slot));
1594  const size_type old_ihash = index_hash(current_hp, hv.hash);
1595  const size_type old_ahash =
1596  alt_index(current_hp, hv.partial, old_ihash);
1597  const size_type new_ihash = index_hash(new_hp, hv.hash);
1598  const size_type new_ahash = alt_index(new_hp, hv.partial, new_ihash);
1599  if ((bucket_i == old_ihash && new_ihash == new_bucket_i) ||
1600  (bucket_i == old_ahash && new_ahash == new_bucket_i)) {
1601  // We're moving the key from the old bucket to the new
1602  // one
1603  Bucket::move_to_bucket(allocator_, old_bucket, slot, new_bucket,
1604  new_bucket_slot++);
1605  // Also update the lock counts, in case we're moving to
1606  // a different lock.
1607  --locks_[lock_ind(bucket_i)].elem_counter();
1608  ++locks_[lock_ind(new_bucket_i)].elem_counter();
1609  } else {
1610  // Check that we don't want to move the new key
1611  assert((bucket_i == old_ihash && new_ihash == old_ihash) ||
1612  (bucket_i == old_ahash && new_ahash == old_ahash));
1613  }
1614  }
1615  }
1616  // Now we can unlock the lock, because all the buckets corresponding
1617  // to it have been unlocked
1618  locks_[start_lock_ind].unlock(LOCK_T());
1619  }
1620  }

1646  // cuckoo_expand_simple will resize the table to at least the given
1647  // new_hashpower. When we're shrinking the table, if the current table
1648  // contains more elements than can be held by new_hashpower, the resulting
1649  // hashpower will be greater than new_hashpower. It needs to take all the
1650  // bucket locks, since no other operations can change the table during
1651  // expansion. Throws libcuckoo_maximum_hashpower_exceeded if we're expanding
1652  // beyond the maximum hashpower, and we have an actual limit.
1653  template <typename LOCK_T, typename AUTO_RESIZE>
1654  cuckoo_status cuckoo_expand_simple(size_type new_hp) {
1655  const auto unlocker = snapshot_and_lock_all<LOCK_T>();
1656  const size_type hp = hashpower();
1657  cuckoo_status st = check_resize_validity<AUTO_RESIZE>(hp, new_hp);
1658  if (st != ok) {
1659  return st;
1660  }
1661  // Creates a new hash table with hashpower new_hp and adds all
1662  // the elements from the old buckets.
1663  cuckoohash_map new_map(hashsize(new_hp) * slot_per_bucket(),
1664  hash_function(), key_eq(), get_allocator());
1665 
1666  parallel_exec(0, hashsize(hp), [this, &new_map](size_type i, size_type end,
1667  std::exception_ptr &eptr) {
1668  try {
1669  for (; i < end; ++i) {
1670  for (size_type j = 0; j < slot_per_bucket(); ++j) {
1671  if (buckets_[i].occupied(j)) {
1672  storage_value_type &kvpair = (buckets_[i].storage_kvpair(j));
1673  new_map.insert(std::move(kvpair.first), std::move(kvpair.second));
1674  }
1675  }
1676  }
1677  } catch (...) {
1678  eptr = std::current_exception();
1679  }
1680  });
1681 
1682  // Swap the current buckets vector with new_map's and set the hashpower.
1683  // This is okay, because we have all the locks, so nobody else should be
1684  // reading from the buckets array. Then the old buckets array will be
1685  // deleted when new_map is deleted. All the locks should be released by
1686  // the unlocker as well.
1687  buckets_.swap(new_map.buckets_);
1688  set_hashpower(new_map.hashpower_);
1689  return ok;
1690  }

1692  // Executes the function over the given range split over num_threads threads
1693  template <typename F>
1694  static void parallel_exec(size_type start, size_type end, F func) {
1695  static const size_type num_threads =
1696  (std::thread::hardware_concurrency() == 0
1697  ? 1
1698  : std::thread::hardware_concurrency());
1699  size_type work_per_thread = (end - start) / num_threads;
1700  std::vector<std::thread,
1701  typename allocator_traits_::template rebind_alloc<std::thread>>
1702  threads(num_threads);
1703  std::vector<
1704  std::exception_ptr,
1705  typename allocator_traits_::template rebind_alloc<std::exception_ptr>>
1706  eptrs(num_threads, nullptr);
1707  for (size_type i = 0; i < num_threads - 1; ++i) {
1708  threads[i] =
1709  std::thread(func, start, start + work_per_thread, std::ref(eptrs[i]));
1710  start += work_per_thread;
1711  }
1712  threads.back() = std::thread(func, start, end, std::ref(eptrs.back()));
1713  for (std::thread &t : threads) {
1714  t.join();
1715  }
1716  for (std::exception_ptr &eptr : eptrs) {
1717  if (eptr) {
1718  std::rethrow_exception(eptr);
1719  }
1720  }
1721  }
