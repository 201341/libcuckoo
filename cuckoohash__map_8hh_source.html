1074  template <typename TABLE_MODE, typename K>

1100  // cuckoo_insert tries to find an empty slot in either of the buckets to
1101  // insert the given key into, performing cuckoo hashing if necessary. It
1102  // expects the locks to be taken outside the function. Before inserting, it
1103  // checks that the key isn't already in the table. cuckoo hashing presents
1104  // multiple concurrency issues, which are explained in the function. The
1105  // following return states are possible:
1106  //
1107  // ok -- Found an empty slot, locks will be held on both buckets after the
1108  // function ends, and the position of the empty slot is returned
1109  //
1110  // failure_key_duplicated -- Found a duplicate key, locks will be held, and
1111  // the position of the duplicate key will be returned
1112  //
1113  // failure_under_expansion -- Failed due to a concurrent expansion
1114  // operation. Locks are released. No meaningful position is returned.
1115  //
1116  // failure_table_full -- Failed to find an empty slot for the table. Locks
1117  // are released. No meaningful position is returned.
1118  template <typename TABLE_MODE, typename K>
1119  table_position cuckoo_insert(const hash_value hv, TwoBuckets &b, K &key) {
1120  int res1, res2;
1121  bucket &b1 = buckets_[b.i1];
1122  if (!try_find_insert_bucket(b1, res1, hv.partial, key)) {
1123  return table_position{b.i1, static_cast<size_type>(res1),
1124  failure_key_duplicated};
1125  }
1126  bucket &b2 = buckets_[b.i2];
1127  if (!try_find_insert_bucket(b2, res2, hv.partial, key)) {
1128  return table_position{b.i2, static_cast<size_type>(res2),
1129  failure_key_duplicated};
1130  }
1131  if (res1 != -1) {
1132  return table_position{b.i1, static_cast<size_type>(res1), ok};
1133  }
1134  if (res2 != -1) {
1135  return table_position{b.i2, static_cast<size_type>(res2), ok};
1136  }
1137 
1138  // We are unlucky, so let's perform cuckoo hashing.
1139  size_type insert_bucket = 0;
1140  size_type insert_slot = 0;
1141  cuckoo_status st = run_cuckoo<TABLE_MODE>(b, insert_bucket, insert_slot);
1142  if (st == failure_under_expansion) {
1143  // The run_cuckoo operation operated on an old version of the table,
1144  // so we have to try again. We signal to the calling insert method
1145  // to try again by returning failure_under_expansion.
1146  return table_position{0, 0, failure_under_expansion};
1147  } else if (st == ok) {
1148  assert(TABLE_MODE() == locked_table_mode() ||
1149  !get_current_locks()[lock_ind(b.i1)].try_lock());
1150  assert(TABLE_MODE() == locked_table_mode() ||
1151  !get_current_locks()[lock_ind(b.i2)].try_lock());
1152  assert(!buckets_[insert_bucket].occupied(insert_slot));
1153  assert(insert_bucket == index_hash(hashpower(), hv.hash) ||
1154  insert_bucket == alt_index(hashpower(), hv.partial,
1155  index_hash(hashpower(), hv.hash)));
1156  // Since we unlocked the buckets during run_cuckoo, another insert
1157  // could have inserted the same key into either b.i1 or
1158  // b.i2, so we check for that before doing the insert.
1159  table_position pos = cuckoo_find(key, hv.partial, b.i1, b.i2);
1160  if (pos.status == ok) {
1161  pos.status = failure_key_duplicated;
1162  return pos;
1163  }
1164  return table_position{insert_bucket, insert_slot, ok};
1165  }
1166  assert(st == failure);
1167  LIBCUCKOO_DBG("hash table is full (hashpower = %zu, hash_items = %zu,"
1168  "load factor = %.2f), need to increase hashpower\n",
1169  hashpower(), size(), load_factor());
1170  return table_position{0, 0, failure_table_full};
1171  }

1190  bool try_find_insert_bucket(const bucket &b, int &slot,

1193  (void)partial;

1197  if (!is_simple && partial != b.partial(i)) {
1198  continue;
1199  }

1202  return false;
1203  }

1227  // run_cuckoo performs cuckoo hashing on the table in an attempt to free up
1228  // a slot on either of the insert buckets, which are assumed to be locked
1229  // before the start. On success, the bucket and slot that was freed up is
1230  // stored in insert_bucket and insert_slot. In order to perform the search
1231  // and the swaps, it has to release the locks, which can lead to certain
1232  // concurrency issues, the details of which are explained in the function.
1233  // If run_cuckoo returns ok (success), then `b` will be active, otherwise it
1234  // will not.
1235  template <typename TABLE_MODE>
1236  cuckoo_status run_cuckoo(TwoBuckets &b, size_type &insert_bucket,
1237  size_type &insert_slot) {
1238  // We must unlock the buckets here, so that cuckoopath_search and
1239  // cuckoopath_move can lock buckets as desired without deadlock.
1240  // cuckoopath_move has to move something out of one of the original
1241  // buckets as its last operation, and it will lock both buckets and
1242  // leave them locked after finishing. This way, we know that if
1243  // cuckoopath_move succeeds, then the buckets needed for insertion are
1244  // still locked. If cuckoopath_move fails, the buckets are unlocked and
1245  // we try again. This unlocking does present two problems. The first is
1246  // that another insert on the same key runs and, finding that the key
1247  // isn't in the table, inserts the key into the table. Then we insert
1248  // the key into the table, causing a duplication. To check for this, we
1249  // search the buckets for the key we are trying to insert before doing
1250  // so (this is done in cuckoo_insert, and requires that both buckets are
1251  // locked). Another problem is that an expansion runs and changes the
1252  // hashpower, meaning the buckets may not be valid anymore. In this
1253  // case, the cuckoopath functions will have thrown a hashpower_changed
1254  // exception, which we catch and handle here.
1255  size_type hp = hashpower();
1256  b.unlock();
1257  CuckooRecords cuckoo_path;
1258  bool done = false;
1259  try {
1260  while (!done) {
1261  const int depth =
1262  cuckoopath_search<TABLE_MODE>(hp, cuckoo_path, b.i1, b.i2);
1263  if (depth < 0) {
1264  break;
1265  }
1266 
1267  if (cuckoopath_move<TABLE_MODE>(hp, cuckoo_path, depth, b)) {
1268  insert_bucket = cuckoo_path[0].bucket;
1269  insert_slot = cuckoo_path[0].slot;
1270  assert(insert_bucket == b.i1 || insert_bucket == b.i2);
1271  assert(TABLE_MODE() == locked_table_mode() ||
1272  !get_current_locks()[lock_ind(b.i1)].try_lock());
1273  assert(TABLE_MODE() == locked_table_mode() ||
1274  !get_current_locks()[lock_ind(b.i2)].try_lock());
1275  assert(!buckets_[insert_bucket].occupied(insert_slot));
1276  done = true;
1277  break;
1278  }
1279  }
1280  } catch (hashpower_changed &) {
1281  // The hashpower changed while we were trying to cuckoo, which means
1282  // we want to retry. b.i1 and b.i2 should not be locked
1283  // in this case.
1284  return failure_under_expansion;
1285  }
1286  return done ? ok : failure;
1287  }

1289  // cuckoopath_search finds a cuckoo path from one of the starting buckets to
1290  // an empty slot in another bucket. It returns the depth of the discovered
1291  // cuckoo path on success, and -1 on failure. Since it doesn't take locks on
1292  // the buckets it searches, the data can change between this function and
1293  // cuckoopath_move. Thus cuckoopath_move checks that the data matches the
1294  // cuckoo path before changing it.
1295  //
1296  // throws hashpower_changed if it changed during the search.
1297  template <typename TABLE_MODE>
1298  int cuckoopath_search(const size_type hp, CuckooRecords &cuckoo_path,
1299  const size_type i1, const size_type i2) {
1300  b_slot x = slot_search<TABLE_MODE>(hp, i1, i2);
1301  if (x.depth == -1) {
1302  return -1;
1303  }
1304  // Fill in the cuckoo path slots from the end to the beginning.
1305  for (int i = x.depth; i >= 0; i--) {
1306  cuckoo_path[i].slot = x.pathcode % slot_per_bucket();
1307  x.pathcode /= slot_per_bucket();
1308  }
1309  // Fill in the cuckoo_path buckets and keys from the beginning to the
1310  // end, using the final pathcode to figure out which bucket the path
1311  // starts on. Since data could have been modified between slot_search
1312  // and the computation of the cuckoo path, this could be an invalid
1313  // cuckoo_path.
1314  CuckooRecord &first = cuckoo_path[0];
1315  if (x.pathcode == 0) {
1316  first.bucket = i1;
1317  } else {
1318  assert(x.pathcode == 1);
1319  first.bucket = i2;
1320  }
1321  {
1322  const auto lock_manager = lock_one(hp, first.bucket, TABLE_MODE());
1323  const bucket &b = buckets_[first.bucket];
1324  if (!b.occupied(first.slot)) {
1325  // We can terminate here
1326  return 0;
1327  }
1328  first.hv = hashed_key(b.key(first.slot));
1329  }
1330  for (int i = 1; i <= x.depth; ++i) {
1331  CuckooRecord &curr = cuckoo_path[i];
1332  const CuckooRecord &prev = cuckoo_path[i - 1];
1333  assert(prev.bucket == index_hash(hp, prev.hv.hash) ||
1334  prev.bucket ==
1335  alt_index(hp, prev.hv.partial, index_hash(hp, prev.hv.hash)));
1336  // We get the bucket that this slot is on by computing the alternate
1337  // index of the previous bucket
1338  curr.bucket = alt_index(hp, prev.hv.partial, prev.bucket);
1339  const auto lock_manager = lock_one(hp, curr.bucket, TABLE_MODE());
1340  const bucket &b = buckets_[curr.bucket];
1341  if (!b.occupied(curr.slot)) {
1342  // We can terminate here
1343  return i;
1344  }
1345  curr.hv = hashed_key(b.key(curr.slot));
1346  }
1347  return x.depth;
1348  }

1350  // cuckoopath_move moves keys along the given cuckoo path in order to make
1351  // an empty slot in one of the buckets in cuckoo_insert. Before the start of
1352  // this function, the two insert-locked buckets were unlocked in run_cuckoo.
1353  // At the end of the function, if the function returns true (success), then
1354  // both insert-locked buckets remain locked. If the function is
1355  // unsuccessful, then both insert-locked buckets will be unlocked.
1356  //
1357  // throws hashpower_changed if it changed during the move.
1358  template <typename TABLE_MODE>
1359  bool cuckoopath_move(const size_type hp, CuckooRecords &cuckoo_path,
1360  size_type depth, TwoBuckets &b) {
1361  if (depth == 0) {
1362  // There is a chance that depth == 0, when try_add_to_bucket sees
1363  // both buckets as full and cuckoopath_search finds one empty. In
1364  // this case, we lock both buckets. If the slot that
1365  // cuckoopath_search found empty isn't empty anymore, we unlock them
1366  // and return false. Otherwise, the bucket is empty and insertable,
1367  // so we hold the locks and return true.
1368  const size_type bucket = cuckoo_path[0].bucket;
1369  assert(bucket == b.i1 || bucket == b.i2);
1370  b = lock_two(hp, b.i1, b.i2, TABLE_MODE());
1371  if (!buckets_[bucket].occupied(cuckoo_path[0].slot)) {
1372  return true;
1373  } else {
1374  b.unlock();
1375  return false;
1376  }
1377  }
1378 
1379  while (depth > 0) {
1380  CuckooRecord &from = cuckoo_path[depth - 1];
1381  CuckooRecord &to = cuckoo_path[depth];
1382  const size_type fs = from.slot;
1383  const size_type ts = to.slot;
1384  TwoBuckets twob;
1385  LockManager extra_manager;
1386  if (depth == 1) {
1387  // Even though we are only swapping out of one of the original
1388  // buckets, we have to lock both of them along with the slot we
1389  // are swapping to, since at the end of this function, they both
1390  // must be locked. We store tb inside the extrab container so it
1391  // is unlocked at the end of the loop.
1392  std::tie(twob, extra_manager) =
1393  lock_three(hp, b.i1, b.i2, to.bucket, TABLE_MODE());
1394  } else {
1395  twob = lock_two(hp, from.bucket, to.bucket, TABLE_MODE());
1396  }
1397 
1398  bucket &fb = buckets_[from.bucket];
1399  bucket &tb = buckets_[to.bucket];
1400 
1401  // We plan to kick out fs, but let's check if it is still there;
1402  // there's a small chance we've gotten scooped by a later cuckoo. If
1403  // that happened, just... try again. Also the slot we are filling in
1404  // may have already been filled in by another thread, or the slot we
1405  // are moving from may be empty, both of which invalidate the swap.
1406  // We only need to check that the hash value is the same, because,
1407  // even if the keys are different and have the same hash value, then
1408  // the cuckoopath is still valid.
1409  if (tb.occupied(ts) || !fb.occupied(fs) ||
1410  hashed_key_only_hash(fb.key(fs)) != from.hv.hash) {
1411  return false;
1412  }
1413 
1414  buckets_.setKV(to.bucket, ts, fb.partial(fs), fb.movable_key(fs),
1415  std::move(fb.mapped(fs)));
1416  buckets_.eraseKV(from.bucket, fs);
1417  if (depth == 1) {
1418  // Hold onto the locks contained in twob
1419  b = std::move(twob);
1420  }
1421  depth--;
1422  }
1423  return true;
1424  }

1504  // slot_search searches for a cuckoo path using breadth-first search. It
1505  // starts with the i1 and i2 buckets, and, until it finds a bucket with an
1506  // empty slot, adds each slot of the bucket in the b_slot. If the queue runs
1507  // out of space, it fails.
1508  //
1509  // throws hashpower_changed if it changed during the search
1510  template <typename TABLE_MODE>
1511  b_slot slot_search(const size_type hp, const size_type i1,
1512  const size_type i2) {
1513  b_queue q;
1514  // The initial pathcode informs cuckoopath_search which bucket the path
1515  // starts on
1516  q.enqueue(b_slot(i1, 0, 0));
1517  q.enqueue(b_slot(i2, 1, 0));
1518  while (!q.full() && !q.empty()) {
1519  b_slot x = q.dequeue();
1520  // Picks a (sort-of) random slot to start from
1521  size_type starting_slot = x.pathcode % slot_per_bucket();
1522  for (size_type i = 0; i < slot_per_bucket() && !q.full(); ++i) {
1523  size_type slot = (starting_slot + i) % slot_per_bucket();
1524  auto lock_manager = lock_one(hp, x.bucket, TABLE_MODE());
1525  bucket &b = buckets_[x.bucket];
1526  if (!b.occupied(slot)) {
1527  // We can terminate the search here
1528  x.pathcode = x.pathcode * slot_per_bucket() + slot;
1529  return x;
1530  }
1531 
1532  // If x has less than the maximum number of path components,
1533  // create a new b_slot item, that represents the bucket we would
1534  // have come from if we kicked out the item at this slot.
1535  const partial_t partial = b.partial(slot);
1536  if (x.depth < MAX_BFS_PATH_LEN - 1) {
1537  b_slot y(alt_index(hp, partial, x.bucket),
1538  x.pathcode * slot_per_bucket() + slot, x.depth + 1);
1539  q.enqueue(y);
1540  }
1541  }
1542  }
1543  // We didn't find a short-enough cuckoo path, so the queue ran out of
1544  // space. Return a failure value.
1545  return b_slot(0, 0, -1);
1546  }

1548  // cuckoo_fast_double will double the size of the table by taking advantage
1549  // of the properties of index_hash and alt_index. If the key's move
1550  // constructor is not noexcept, we use cuckoo_expand_simple, since that
1551  // provides a strong exception guarantee.
1552  template <typename TABLE_MODE, typename AUTO_RESIZE>
1553  cuckoo_status cuckoo_fast_double(size_type current_hp) {
1554  if (!std::is_nothrow_move_constructible<key_type>::value ||
1555  !std::is_nothrow_move_constructible<mapped_type>::value) {
1556  LIBCUCKOO_DBG("%s", "cannot run cuckoo_fast_double because key-value"
1557  " pair is not nothrow move constructible");
1558  return cuckoo_expand_simple<TABLE_MODE, AUTO_RESIZE>(current_hp + 1);
1559  }
1560  const size_type new_hp = current_hp + 1;
1561  auto all_locks_manager = snapshot_and_lock_all(TABLE_MODE());
1562  cuckoo_status st = check_resize_validity<AUTO_RESIZE>(current_hp, new_hp);
1563  if (st != ok) {
1564  return st;
1565  }

1567  // We must re-hash the table, moving items in each bucket to a different
1568  // one. The hash functions are carefully designed so that when doubling the
1569  // number of buckets, each element either stays in its existing bucket or
1570  // goes to exactly one new bucket. This means we can re-hash each bucket in
1571  // parallel. We create a new empty buckets container and move all the
1572  // elements from the old container to the new one.
1573  buckets_t new_buckets(new_hp, get_allocator());
1574  parallel_exec(
1575  0, hashsize(current_hp),
1576  [this, &new_buckets, current_hp, new_hp](size_type start, size_type end,
1577  std::exception_ptr &eptr) {
1578  try {
1579  move_buckets(new_buckets, current_hp, new_hp, start, end);
1580  } catch (...) {
1581  eptr = std::current_exception();
1582  }
1583  });

1586  // hashpower so that other threads don't grab the new hashpower and the old
1587  // locks
1588  maybe_resize_locks(1UL << new_hp);
1589  // Swap the old and new buckets. The old bucket data will be destroyed when
1590  // the function exits
1591  buckets_.swap(new_buckets);
1592  return ok;
1593  }

1595  void move_buckets(buckets_t &new_buckets, size_type current_hp,
1596  size_type new_hp, size_type start_ind, size_type end_ind) {
1597  for (size_type old_bucket_ind = start_ind; old_bucket_ind < end_ind;
1598  ++old_bucket_ind) {
1599  // By doubling the table size, the index_hash and alt_index of
1600  // each key got one bit added to the top, at position
1601  // current_hp, which means anything we have to move will either
1602  // be at the same bucket position, or exactly
1603  // hashsize(current_hp) later than the current bucket
1604  bucket &old_bucket = buckets_[old_bucket_ind];
1605  const size_type new_bucket_ind = old_bucket_ind + hashsize(current_hp);
1606  size_type new_bucket_slot = 0;
1607 
1608  // For each occupied slot, either move it into its same position in the
1609  // new buckets container, or to the first available spot in the new
1610  // bucket in the new buckets container.
1611  for (size_type old_bucket_slot = 0; old_bucket_slot < slot_per_bucket();
1612  ++old_bucket_slot) {
1613  if (!old_bucket.occupied(old_bucket_slot)) {
1614  continue;
1615  }
1616  const hash_value hv = hashed_key(old_bucket.key(old_bucket_slot));
1617  const size_type old_ihash = index_hash(current_hp, hv.hash);
1618  const size_type old_ahash =
1619  alt_index(current_hp, hv.partial, old_ihash);
1620  const size_type new_ihash = index_hash(new_hp, hv.hash);
1621  const size_type new_ahash = alt_index(new_hp, hv.partial, new_ihash);
1622  size_type dst_bucket_ind, dst_bucket_slot;
1623  if ((old_bucket_ind == old_ihash && new_ihash == new_bucket_ind) ||
1624  (old_bucket_ind == old_ahash && new_ahash == new_bucket_ind)) {
1625  // We're moving the key to the new bucket
1626  dst_bucket_ind = new_bucket_ind;
1627  dst_bucket_slot = new_bucket_slot++;
1628  } else {
1629  // We're moving the key to the old bucket
1630  assert((old_bucket_ind == old_ihash && new_ihash == old_ihash) ||
1631  (old_bucket_ind == old_ahash && new_ahash == old_ahash));
1632  dst_bucket_ind = old_bucket_ind;
1633  dst_bucket_slot = old_bucket_slot;
1634  }
1635  new_buckets.setKV(dst_bucket_ind, dst_bucket_slot++,
1636  old_bucket.partial(old_bucket_slot),
1637  old_bucket.movable_key(old_bucket_slot),
1638  std::move(old_bucket.mapped(old_bucket_slot)));
1639  }
1640  }
1641  }

1648  template <typename AUTO_RESIZE>
1649  cuckoo_status check_resize_validity(const size_type orig_hp,
1650  const size_type new_hp) {
1651  const size_type mhp = maximum_hashpower();
1652  if (mhp != LIBCUCKOO_NO_MAXIMUM_HASHPOWER && new_hp > mhp) {
1653  throw libcuckoo_maximum_hashpower_exceeded(new_hp);
1654  }
1655  if (AUTO_RESIZE::value && load_factor() < minimum_load_factor()) {
1656  throw libcuckoo_load_factor_too_low(minimum_load_factor());
1657  }
1658  if (hashpower() != orig_hp) {
1659  // Most likely another expansion ran before this one could grab the
1660  // locks
1661  LIBCUCKOO_DBG("%s", "another expansion is on-going\n");

1667  // When we expand the contanier, we may need to expand the locks array, if
1668  // the current locks array is smaller than the maximum size and also smaller
1669  // than the number of buckets in the upcoming buckets container. In this
1670  // case, we grow the locks array to the smaller of the maximum lock array
1671  // size and the bucket count. This is done by allocating an entirely new lock
1672  // container, taking all the locks, copying over the counters, and then
1673  // finally adding it to the end of `all_locks_`, thereby designating it the
1674  // "current" locks container. It is the responsibility of the caller to
1675  // unlock all locks taken, including the new locks, whenever it is done with
1676  // them, so that old threads can resume and potentially re-start.
1677  void maybe_resize_locks(size_type new_bucket_count) {
1678  locks_t &current_locks = get_current_locks();
1679  if (!(current_locks.size() < kMaxNumLocks &&
1680  current_locks.size() < new_bucket_count)) {
1681  return;
1682  }
1683 
1684  locks_t new_locks(std::min(size_type(kMaxNumLocks), new_bucket_count),
1685  spinlock(), get_allocator());
1686  for (spinlock &lock : new_locks) {
1687  lock.lock();
1688  }
1689  assert(new_locks.size() > current_locks.size());
1690  std::copy(current_locks.begin(), current_locks.end(), new_locks.begin());
1691  all_locks_.emplace_back(std::move(new_locks));
1692  }

1694  // cuckoo_expand_simple will resize the table to at least the given
1695  // new_hashpower. When we're shrinking the table, if the current table
1696  // contains more elements than can be held by new_hashpower, the resulting
1697  // hashpower will be greater than `new_hp`. It needs to take all the bucket
1698  // locks, since no other operations can change the table during expansion.
1699  // Throws libcuckoo_maximum_hashpower_exceeded if we're expanding beyond the
1700  // maximum hashpower, and we have an actual limit.
1701  template <typename TABLE_MODE, typename AUTO_RESIZE>
1702  cuckoo_status cuckoo_expand_simple(size_type new_hp) {
1703  auto all_locks_manager = snapshot_and_lock_all(TABLE_MODE());
1704  const size_type hp = hashpower();
1705  cuckoo_status st = check_resize_validity<AUTO_RESIZE>(hp, new_hp);
1706  if (st != ok) {
1707  return st;
1708  }
1709  // Creates a new hash table with hashpower new_hp and adds all
1710  // the elements from the old buckets.
1711  cuckoohash_map new_map(hashsize(new_hp) * slot_per_bucket(),
1712  hash_function(), key_eq(), get_allocator());
1713 
1714  parallel_exec(0, hashsize(hp), [this, &new_map](size_type i, size_type end,
1715  std::exception_ptr &eptr) {
1716  try {
1717  for (; i < end; ++i) {
1718  for (size_type j = 0; j < slot_per_bucket(); ++j) {
1719  if (buckets_[i].occupied(j)) {
1720  new_map.insert(buckets_[i].movable_key(j),
1721  std::move(buckets_[i].mapped(j)));
1722  }
1723  }
1724  }
1725  } catch (...) {
1726  eptr = std::current_exception();
1727  }
1728  });
1729 
1730  // Swap the current buckets containers with new_map's. This is okay,
1731  // because we have all the locks, so nobody else should be reading from the
1732  // buckets array. Then the old buckets array will be deleted when new_map
1733  // is deleted. We also resize the locks array if necessary.
1734  maybe_resize_locks(new_map.bucket_count());
1735  buckets_.swap(new_map.buckets_);
1736  return ok;
1737  }

1739  // Executes the function over the given range split over num_threads threads
1740  template <typename F>
1741  void parallel_exec(size_type start, size_type end, F func) {
1742  static const size_type num_threads =
1743  std::max(std::thread::hardware_concurrency(), 1U);
1744  size_type work_per_thread = (end - start) / num_threads;
1745  std::vector<std::thread, rebind_alloc<std::thread>> threads(
1746  get_allocator());
1747  threads.reserve(num_threads);
1748  std::vector<std::exception_ptr, rebind_alloc<std::exception_ptr>> eptrs(
1749  num_threads, nullptr, get_allocator());
1750  for (size_type i = 0; i < num_threads - 1; ++i) {
1751  threads.emplace_back(func, start, start + work_per_thread,
1752  std::ref(eptrs[i]));
1753  start += work_per_thread;
1754  }
1755  threads.emplace_back(func, start, end, std::ref(eptrs.back()));
1756  for (std::thread &t : threads) {
1757  t.join();
1758  }
1759  for (std::exception_ptr &eptr : eptrs) {
1760  if (eptr) {
1761  std::rethrow_exception(eptr);
1762  }
1763  }
1764  }
